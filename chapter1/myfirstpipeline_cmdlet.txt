-------Powershell commands-----------------------------------------------------------
-----Create Resource Group----------------------------------------

New-AzResourceGroup -Name rg-adftutorial -Location "Central US"


-----------------------------------------------------------------
------------Create Resource Storage Account----------------------

New-AzResourceGroupDeployment `
    -ResourceGroupName rg-adftutorial `
    -TemplateFile .\myfirstpipeline.json `
    -TemplateParameterFile .\myfirstpipeline.parameters.json `
    -Verbose


-----------------------------------------------------------------
------------Delete Resource Group--------------------------------

Get-AzureRmResourceGroup -Name "rg-adftutorial" | Remove-AzureRmResourceGroup -Verbose -Force
